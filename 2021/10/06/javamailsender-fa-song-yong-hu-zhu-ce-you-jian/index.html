<script>
    new Vue({
        el: '#app',
        data(){
            return{
                outerVisible: false,
                innerVisible: false,
                dynamicValidateForm: {
                    email: ''
                }
            }
        },
        methods:{
            sendCaptcha(){
                var email = document.getElementById('email').value;
                var reg = /^([a-zA-Z]|[0-9])(\w|\-)+@[a-zA-Z0-9]+\.([a-zA-Z]{2,4})$/;
                if (!reg.test(email)){
                    document.getElementById("userCue").innerHTML="<font color='red'><b>邮箱格式错误</b></font>"
                    return false
                }else {
                    var loading = this.$loading({
                        text: '正在加载',
                        spinner: 'el-icon-loading',
                        background: 'rgba(0,0,0,.8)'
                    })
                    axios.post('sendCaptcha?email='+email,email).then(res => {
                        loading.close()
                        /*发送验证码开始*/
                        // 定时器,60s发一次验证码
                        var time = 60;
                        var btn = document.getElementById('sendCaptcha')
                        btn.disabled = true;
                        btn.style.cursor="not-allowed";
                        var timer = setInterval(function (){
                            if (time == 0){
                                clearInterval(timer);
                                btn.disabled = false;
                                btn.style.cursor="pointer";
                                btn.innerHTML = '发送邮箱验证码';
                                time = 60;
                                return false
                            }
                            btn.innerHTML = '还剩'+time+'s再次发送';
                            time--;
                        },1000)
                        this.$notify.success({
                            title: '成功',
                            message: '验证码已成功发送至您的邮箱，请及时查收5min内有效',
                            duration: 5000
                        })
                        document.getElementById('userCue').innerHTML="<font color='green'><b>验证码已发送至邮箱,请注意查收</b></font>"
                    }).catch(err => {
                        loading.close()
                        this.$notify.error({
                            title: '错误',
                            message: '邮箱已被注册，如非本人，请联系管理员！',
                            duration: 5000
                        })
                    })

    }
            },
            // 重置密码邮箱验证
            resetPasswordCheckEmail(email) {
                var reg = /^([a-zA-Z]|[0-9])(\w|\-)+@[a-zA-Z0-9]+\.([a-zA-Z]{2,4})$/;
                if (!reg.test(email)){
                    this.$message({
                        type: 'error',
                        message: '邮箱格式错误,请输入正确邮箱!'
                    })
                    return false;
                };
                axios.get('resetPasswordCheckEmail?email='+email).then(res =>{
                    var loading = this.$loading({
                        text: '正在加载',
                        spinner: 'el-icon-loading',
                        background: 'rgba(0,0,0,.8)'
                    });
                    var Message = res.data.msg;
                    var isSuccess = res.data.data;
                    if (isSuccess){
                        this.$notify.success({
                            title: '成功',
                            message: '邮箱正确，将发送链接至对应邮箱',
                            duration: 5000
                        });
                        axios.post('sendResetPasswordLink?email='+email).then(res =>{
                            loading.close();
                            this.$notify.success({
                                title: '成功',
                                message: '链接已成功发送至邮箱，请注意查收！30min内有效！',
                                duration: 5000
                            });
                            this.outerVisible = false;
                        })
                    }else{
                        this.$message({
                            type: 'error',
                            message: Message
                        })
                    }
                })
            }
        }
    })

   /**
    * 发送邮件
    * @param email
    * @param request
    * @return
    */
   @PostMapping("/sendCaptcha")
   public ResponseEntity<String> sendCaptcha(
         @RequestParam(value="email") String email,
         HttpServletRequest request
   ){
      /*首先校验邮箱唯一性*/
      //    校验注册邮箱唯一
      Student findByEmail =this.studentService.findByEmail(email);
      if (findByEmail != null){
         return ResponseEntity.status(HttpStatus.BAD_REQUEST).build();
      }
      int captcha = (int) ((Math.random() * 9 + 1) * 100000);
      // 发送邮件
      SimpleMailMessage message = new SimpleMailMessage();
      message.setSubject("欢迎注册校园二手信息共享平台，请查收验证码");
      message.setText("尊敬的用户,您好:\n"
            + "\n本次请求的邮件验证码为:" + captcha + ",本验证码5分钟内有效，请及时输入。（请勿泄露此验证码）\n"
            + "\n如非本人操作，请忽略该邮件。\n(这是一封自动发送的邮件，请不要直接回复）");
      message.setFrom(emailSendForm);
      message.setTo(email);
      javaMailSender.send(message);
      //将验证码存入session
      request.getSession().setAttribute("user_register",captcha);
      //将接收验证码的邮箱存入session
      request.getSession().setAttribute("user_register_email",email);
      log.info("验证码发送成功,user_register" + ",value="+ captcha);
      return ResponseEntity.ok().build();
   }

   /**
    * 发送重置密码连接
    * @param email
    * @param request
    * @return
    * @throws MessagingException
    * @throws ParseException
    */
   @PostMapping("/sendResetPasswordLink")
   @ResponseBody
   public ResponseEntity<String> sendResetPasswordLink(
         @RequestParam(value = "email")String email,
         HttpServletRequest request
   ) throws MessagingException, ParseException {
      /*项目的服务器地址*/
      String serverAddress = request.getScheme() + "://" + request.getServerName() + ":" + request.getServerPort();
      /*根据email查询用户相对应的学号*/
      Student student = this.studentService.findByEmail(email);
      String sn = student.getSn();

      String resetPasswordStartTime = StringUtil.getFormatterDate(new Date(), "yyyy-MM-dd HH:mm:ss");
      Date dateStart = StringUtil.getFormatterDateFromString(resetPasswordStartTime, "yyyy-MM-dd HH:mm:ss");
      long operateTimeStart = dateStart.getTime();
      /*将开始申请重置密码的操作时间存入session
      * 根据邮箱，找寻对应的用户学号存入session
      * */
      request.getSession().setAttribute("resetPassword_operateTime",operateTimeStart);
      request.getSession().setAttribute("resetPassword_sn",sn);

      /*重置密码接口链接*/
      String resetPasswordLink = serverAddress + "/home/index/resetPasswordByEmailIndex";
      /*发送重置密码链接的邮件*/
      MimeMessage mimeMessage = mailSender.createMimeMessage();
      MimeMessageHelper helper = new MimeMessageHelper(mimeMessage);
      helper.setFrom(emailSendForm);
      helper.setTo(email);
      helper.setSubject("智易平台密码重置");
      helper.setText("您正在为账号"+email.toString()+"进行密码重置操作。该链接30min内有效，请即时更改。点此链接进行密码重置"+resetPasswordLink);
      mailSender.send(mimeMessage);
      return ResponseEntity.ok().build();
   }

   @PostMapping("/resetPassword")
   @ResponseBody
   public Result<Boolean> resetPassword(
         HttpServletRequest request,
         @RequestParam(value = "sn")String sn,
         @RequestParam(value = "password")String password,
         @RequestParam(value = "passwordAgain")String passwordAgina,
         @RequestParam(value = "captcha")String captcha) throws ParseException {
      /*获取session中之前存入的sn学号和开始申请重置密码的操作时间operateTime*/
      String resetPassword_sn = request.getSession().getAttribute("resetPassword_sn").toString();
      long resetPassword_operateTime = (long) request.getSession().getAttribute("resetPassword_operateTime");
      /*获取session中验证码*/
      String captchaSession =(String) request.getSession().getAttribute("admin_login");
      /*获取当前操作时间*/
      String resetPassword_end = StringUtil.getFormatterDate(new Date(),"yyyy-MM-dd HH:mm:ss");
      Date resetPassword_endT = StringUtil.getFormatterDateFromString(resetPassword_end,"yyyy-MM-dd HH:mm:ss");
      long resetPassword_endTime = resetPassword_endT.getTime();
      /*计算从开始申请重置密码到开始重置的时间差min*/
      int minutes = (int) ((resetPassword_endTime - resetPassword_operateTime) / (1000 * 60));
      /*判断操作是否超时*/
      if (!(minutes >= 0 && minutes <= 30)){
         return Result.error(CodeMsg.USER_PASSWORD_RESET_TIMEOUT);
      }
      /*判断学号是否与邮箱对应*/
      if (!resetPassword_sn.equals(sn)){
         return Result.error(CodeMsg.USER_PASSWORD_RESET_SN_ERROR);
      }
      /*判断两个密码是否一致*/
      if (!password.equals(passwordAgina)){
         return Result.error(CodeMsg.USER_PASSWORD_RESET_NOTSAME);
      }

      /*判断验证码是否一致*/
      if (!(captcha.toUpperCase()).equals(captchaSession.toUpperCase())){
         return Result.error(CodeMsg.CAPTCHA_ERROR);
      }
      /*根据学号重置密码*/
      this.studentService.updateStuPasswordBySn(sn,password);

      return Result.success(true);
   }

      JSONObject regionRealAddress = this.studentService.getRegionRealAddress(ipAddress, baiduAK);

      HashMap<Object, Object> map = new HashMap<>();
      map.put("nowProvince",regionRealAddress.get("province"));
      map.put("nowCity",regionRealAddress.get("city"));
      return map;
   }
