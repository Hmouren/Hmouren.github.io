使用

  • Spring Data Redis提供了一个工具类： RedisTemplate。里面 封装了对于Redis的五种数据结构的各种操作，包括如下：
  1. redis Template.opsForValue()：操作字符串
  2. redis.Template.opsForHash(): 操作hash
  3. redis.Template.opsForList():操作list
  4. redis.Template.opsForSet():操作set
  5. redis.Template.opsForZSet():操作zset

其他一些通用命令，如expire可通过redisTemplate.xx来直接调用

  • 5种结构
  1. String：等同于java中的，Map<String,String>
  2. list:等同于java中的，Map<String,List>
  3. set:等同于java中的，Map<String,Set>
  4. sort_set:等同于java中的，可排序的set
  5. hash:等同于java中的，Map<String,Map<String,String>>

export default {
  name: 'Register',
  data () {
    var validatePass = (rule, value, callback) => {
      if (value === '') {
        callback(new Error('请输入密码'))
      } else {
        if (value !== '') {
          var reg = /(?=.*[0-9])(?=.*[a-zA-Z]).{8,30}/
          if (!reg.test(value)) {
            callback(new Error('密码中必须包含字母、数字，至少8位，至多30'))
          }
        }
        callback()
      }
    }
    var validatePass2 = (rule, value, callback) => {
      if (value === '') {
        callback(new Error('请再次输入密码'))
      } else if (value !== this.ruleForm.pass) {
        callback(new Error('两次输入密码不一致!'))
      } else {
        callback()
      }
    }
    var checkUserName = (rule, value, callback) => {
      if (value === '') {
        callback(new Error('请输入用户名'))
      }
      callback()
    }
    var checkStuNum = (rule, value, callback) => {
      if (value === '') {
        callback(new Error('请输入学号'))
      }
      callback()
    }
    var checkPhone = (rule, value, callback) => {
      if (value === '') {
        callback(new Error('手机号不可为空'))
      } else {
        if (value !== '') {
          var reg = /^1[3456789]\d{9}$/
          if (!reg.test(value)) {
            callback(new Error('请输入有效的手机号码'))
          }
        }
        callback()
      }
    }
    var checkEmail = (rule, value, callback) => {
      if (value === '') {
        callback(new Error('邮箱不可为空'))
      } else {
        if (value !== '') {
          var reg = /^\w+((.\w+)|(-\w+))@[A-Za-z0-9]+((.|-)[A-Za-z0-9]+).[A-Za-z0-9]+$/
          if (!reg.test(value)) {
            callback(new Error('邮箱格式不正确，请输入有效的邮箱账号'))
          }
        }
      }
    }
    return {
      ruleForm: {
        pass: '',
        checkPass: '',
        userName: '',
        stuNum: '',
        sex: '1',
        phone: '',
        email: '',
        code: '',
        options: [{

          label: '湖北医药学院',

          }]

        ],
        checkPass: [
          { required: true, validator: validatePass2, trigger: 'blur' }
        ],
        userName: [
          { required: true, validator: checkUserName, trigger: 'blur' }
        ],
        stuNum: [
          {required: true, validator: checkStuNum, trigger: 'blur'}
        ],
        phone: [
          {required: true, validator: checkPhone, trigger: 'blur'}
        ],
        email: [
          {required: true, validator: checkEmail, trigger: 'blur'}
        ]

    }

import ch.qos.logback.core.util.TimeUtil;

@Controller
@RequestMapping("/WEhbmu")
public class UserController {

    @Autowired
    private UserService userService;

    @Autowired
    private JavaMailSender javaMailSender;

    @Autowired
    private StringRedisTemplate redisTemplate;

    //    定义前缀信息区分邮箱有验证码
    private static final String KEY_PREFIX = "user:verify:";


    /**
     * @param
     * @return 实现用户注册功能
     */
    @PostMapping("/register")
    @CrossOrigin
    @ResponseBody
    public String saveUser(
            @RequestParam(value = "userName") String userName,
            @RequestParam(value = "pass") String pass,
            @RequestParam(value = "stuNum") Integer stuNum,
            @RequestParam(value = "sex") Integer sex,
            @RequestParam(value = "phone") String phone,
            @RequestParam(value = "code") String code,
            @RequestParam(value = "email") String email

    ) {
        List<User> users = this.userService.queryAllByStuNum(stuNum);
        if (CollectionUtils.isEmpty(users)) {
            User user = new User();
            user.setUserName(userName);
            user.setPass(pass);
            user.setStuNum(stuNum);
            user.setPhone(phone);

//            获取redis保存的验证码校验
            String valCode = this.redisTemplate.opsForValue().get("KEY_PREFIX:" + email).toString();
            if (valCode.equals(code)) {
                if (sex == 1) {
                    user.setSex("男");
                }
                user.setSex("女");

                user.setSchool("湖北医药学院");
                user.setArea("校本部");
                user.setStuPic("http://125");

                this.userService.saveUser(user);
                return "success";
            }
            return "codeError";
        }
        return "error";
    }

    /**
     * @param email 利用smtp实现邮箱验证码
     */
    @RequestMapping("/sendEmail")
    @ResponseBody
    public void sendEmail(@RequestParam(value = "email") String email) {
        // 测试邮箱验证
        int s = (int) ((Math.random() * 9 + 1) * 100000);
        String code = String.valueOf(s);
        SimpleMailMessage message = new SimpleMailMessage();
        message.setSubject("欢迎注册校园二手交易平台，请查收验证码");
        message.setText("尊敬的用户,您好:\n"
                + "\n本次请求的邮件验证码为:" + code + ",本验证码2分钟内有效，请及时输入。（请勿泄露此验证码）\n"
                + "\n如非本人操作，请忽略该邮件。\n(这是一封自动发送的邮件，请不要直接回复）");
        message.setFrom("wehbmu@qq.com");
        message.setTo(email);
        javaMailSender.send(message);

//向redis里存入数据并设置过期事件2min
        this.redisTemplate.opsForValue().set("KEY_PREFIX:" + email, code, 2, TimeUnit.MINUTES);
    }

    /**
     * @param stuNum
     * @param password
     * @return 实现登录
     */
    @PostMapping("/login")
    @ResponseBody
    @CrossOrigin
    public String login(
            @RequestParam(value = "stuNum") String stuNum,
            @RequestParam(value = "password") String password
    ) {

        List<User> users = this.userService.queryUserByUserNameAndPassword(stuNum, password);
        if (CollectionUtils.isEmpty(users)) {
            return "error";
        }
        return "success";
    }

}
